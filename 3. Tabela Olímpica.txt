*Criando o banco de dados
----------------------------------------------------------------
create database Olimpiadas;
use Olimpiadas;
----------------------------------------------------------------
*Criando a tabela "Atleta"
----------------------------------------------------------------
create table Atleta(idAtleta int primary key auto_increment, nome varchar(40), modalidade varchar(40), pais varchar(30), cidadeAtleta varchar(40), qtdMedalhaBronze int(10), qtdMedalhaPrata int(10), qtdMedalhaOuro int(10), ano char(4), dataDeNascimento varchar(10));
-----------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
*Mostrar só as colunas das tabelas
-----------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
Desc nome-da-tabela;
*ou
SHOW COLUMNS FROM nome-da-tabela;

+------------------+-------------+------+-----+---------+----------------+
| Field            | Type        | Null | Key | Default | Extra          |
+------------------+-------------+------+-----+---------+----------------+
| idAtleta         | int         | NO   | PRI | NULL    | auto_increment |
| nome             | varchar(40) | YES  |     | NULL    |                |
| modalidade       | varchar(40) | YES  |     | NULL    |                |
| pais             | varchar(30) | YES  |     | NULL    |                |
| cidadeAtleta     | varchar(40) | YES  |     | NULL    |                |
| qtdMedalhaBronze | int         | YES  |     | NULL    |                |
| qtdMedalhaPrata  | int         | YES  |     | NULL    |                |
| qtdMedalhaOuro   | int         | YES  |     | NULL    |                |
| ano              | char(4)     | YES  |     | NULL    |                |
| dataDeNascimento | varchar(10) | YES  |     | NULL    |                |
